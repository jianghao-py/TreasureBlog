### 冒泡排序

<br>

```javascript
    function filter(arry){
            let count = null;

            for(i = 0;i<arry.length - 1;i++){
                for(j = 0;j < arry.length - i - 1;j++){
                    if(arry[j].age > arry[j+1].age){
                        count = arry[j];
                        arry[j] = arry[j+1];
                        arry[j+1] = count
                    }   
                }
            }
        return arry;
    }

    console.log(filter(arry));
```
